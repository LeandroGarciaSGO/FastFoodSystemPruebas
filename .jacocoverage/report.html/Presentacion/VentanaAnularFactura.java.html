    private boolean validarCampos(){
        try{
            Long.parseLong(jTextFieldNumFactura.getText());
            if(jTextFieldNumFactura.getText().length() <= 0){
                JOptionPane.showMessageDialog(this, "ERROR: El Numero de Factura No Debe Ser Vacio", "FastFoodSystem", JOptionPane.ERROR_MESSAGE);
                return false;
            }
            Integer.parseInt(jTextFieldNumFactura.getText());
            }
        catch(NumberFormatException e){
            JOptionPane.showMessageDialog(this, "ERROR: El Numero de Factura Debe Ser Numerico", "FastFoodSystem", JOptionPane.ERROR_MESSAGE);
            return false;
        }
        return true;
        
    }   
    
    
    private void jButtonBuscarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButtonBuscarActionPerformed
        // TODO add your handling code here:
        Facturacion res =  null;
        if (validarCampos()) {
            OperacionesFacturacion OF = new OperacionesFacturacion();
            
           
            try {
                res = OF.buscarFactura(Integer.parseInt(jTextFieldNumFactura.getText()));
            } catch (SQLException ex) {
                Logger.getLogger(VentanaAnularFactura.class.getName()).log(Level.SEVERE, null, ex);
            } catch (ClassNotFoundException ex) {
                Logger.getLogger(VentanaAnularFactura.class.getName()).log(Level.SEVERE, null, ex);
            }
            if(res!=null){
                if(!res.isEstado()){
                    JOptionPane.showMessageDialog(this, "ERROR: La Factura Ya Esta Anulada", "FastFoodSystem", JOptionPane.ERROR_MESSAGE);
            
                }else{
            VentanaConfirmacion VC = new VentanaConfirmacion();
            VC.setFactura(res);
            VC.setDatosUsuario(datosUsuario);
            VC.cargarDatos();
            VC.setVisible(true);
            this.dispose();}
//            if(!res){

//            }
            }else{
                 JOptionPane.showMessageDialog(this, "ERROR: La Factura Inexistente", "FastFoodSystem", JOptionPane.ERROR_MESSAGE);
            }
//            if(res == 0){

//            }
        }
